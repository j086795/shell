#!/bin/bash
#Este programa mostra o uso do comando if elif fi
  echo "Selecione uma opção:"
  echo "1 - Exibir data e hora do sistema"
  echo "2 - Exibir resultado da divisão 10/2"
  echo "3 - Exibir uma mensagem"

  read opcao;

  if [ $opcao -eq 1 ];then
   date
   echo
   $SHELL
  elif [ $opcao -eq 2 ];
  then
   result=$((10/2))
   echo "divisão de 10/2 = $result"
   echo
   $SHELL
  elif [ $opcao -eq 3 ];
  then
   echo "Informe seu nome:"
   read nome;
   echo "Bem vindo ao mundo do shell script, $nome!"
   echo
   $SHELL
  fi
